<script>
function mostrar(){

let campos = document.getElementsByTagName("input");
let lista = document.getElementById("lista");

for(let campo of campos){

    if(campo.value != ""){
        let item = document.createElement("li");
        item.innerHTML = campo.value;
        lista.appendChild(item);

        campo.value = "";
    }

}

}
</script>
